function doGet(e) { const props = PropertiesService.getScriptProperties(); const count = parseInt(props.getProperty('callCount') || '0', 10); const lastIncrement = props.getProperty('lastIncrement'); let timeDisplay = ''; if (lastIncrement) { const date = new Date(parseInt(lastIncrement)); const now = new Date(); const diffMinutes = Math.floor((now - date) / (1000 * 60)); if (diffMinutes < 1) { timeDisplay = 'Just now'; } else if (diffMinutes < 60) { timeDisplay = `${diffMinutes}m ago`; } else if (diffMinutes < 1440) { const hours = Math.floor(diffMinutes / 60); timeDisplay = `${hours}h ago`; } else { timeDisplay = date.toLocaleDateString() + ' ' + date.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}); } } else { timeDisplay = 'No calls yet'; } const html = `
Transfer Count
${count}
Last: ${timeDisplay}
`; return HtmlService .createHtmlOutput(html) .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // 2) Increment counter and store timestamp function doPost(e) { const props = PropertiesService.getScriptProperties(); const current = parseInt(props.getProperty('callCount') || '0', 10); const updated = current + 1; const timestamp = new Date().getTime(); props.setProperties({ 'callCount': updated.toString(), 'lastIncrement': timestamp.toString() }); return ContentService .createTextOutput(JSON.stringify({ count: updated, timestamp: timestamp, timeFormatted: new Date(timestamp).toISOString() })) .setMimeType(ContentService.MimeType.JSON); } // 3) Reset counter and clear timestamp function resetCounter() { const props = PropertiesService.getScriptProperties(); props.setProperties({ 'callCount': '0', 'lastIncrement': null }); return ContentService .createTextOutput(JSON.stringify({ count: 0, message: 'Counter and timestamp reset' })) .setMimeType(ContentService.MimeType.JSON); }
